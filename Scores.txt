Loading leaderboard (read all lines, split by " | ", sort by score)

def load_leaderboard():
    scores = []
    with open("scores.txt", "r") as file:
        for line in file:
            name, grade, score = line.strip().split(" | ")
            scores.append((name, grade, int(score)))
    # sort by score descending
    scores.sort(key=lambda x: x[2], reverse=True)
    return scores


Displaying leaderboard

for rank, (name, grade, score) in enumerate(load_leaderboard(), start=1):
    print(f"{rank}. {name} ({grade}) - {score} pts")
